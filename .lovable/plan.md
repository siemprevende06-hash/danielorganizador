const allBlocks = blocks.sort((a, b) => {
  const [aH, aM] = a.startTime.split(':').map(Number);
  const [bH, bM] = b.startTime.split(':').map(Number);
  return (aH * 60 + aM) - (bH * 60 + bM);
});
